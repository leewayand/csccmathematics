\begin{exercise}





Let 


\[
m(t) = -4 \ln(-x+7)  + 5
\]





\begin{question}



The zero of $m$ is


\[
\answer{ 7 - e^{\frac{5}{4}} }
\]


\end{question}









\begin{question}



$m$ is positive on 


\[
\left( \answer{ 7 - e^{\frac{5}{4}} }, 7 \right)
\]


\end{question}








\begin{question}



$m$ is negative on 


\[
\left( \answer{-\infty}, \answer{ 7 - e^{\frac{5}{4}} }  \right)
\]


\end{question}









\end{exercise}
