\begin{exercise}

Let $a$ be a real number such that $2 < a < 5$. \\

Select all expressions equivalent to $|2-a|$. \\


\begin{selectAll}
\choice {$2 - a$}   
\choice {$2 + a$}   
\choice [correct]{$-(2 - a)$}   
\choice {$-2 - a$}  
\choice [correct]{$a - 2$}  
\end{selectAll}


\end{exercise}
