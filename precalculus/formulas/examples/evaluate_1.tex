\begin{exercise}

Let $f(x) = \frac{6x-1}{4 + 2x}$ with its natural domain. \\

Evaluate the following.  (Use DNE for Does not Exist) \\


\begin{align*}
f(0) &= \answer{-\frac{1}{4}} \\
f\left(\frac{1}{2}\right) &= \answer{\frac{2}{5}} \\
f(-2)    &= \answer{DNE} \\
f\left(\frac{1}{6}\right) &= \answer{0} \\
f(-1) &= \answer{-\frac{7}{2}}  \\
f(1) &= \answer{\frac{5}{6}}
\end{align*}


\end{exercise}












\begin{exercise}

Evaluate the following.  (Use DNE for Does not Exist) \\

\end{exercise}
