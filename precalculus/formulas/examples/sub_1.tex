\begin{question}
Let $f(x) = 5 x + 7$ \\

Let $a$ and $h$ be real numbers. \\

Which expression is equivalent to $f(a+h)$?


\begin{multipleChoice}
\choice {$5a+7+h$}
\choice {$5a + h + 7$}
\choice [correct]{$5(a+h) + 7$}
\choice {$5x + 7 + a + h$}
\end{multipleChoice}

\end{question}










\begin{question}
Let $G(t) = 8t^2 - 5$ \\

Let $a$ and $h$ be real numbers. \\

Which expression is equivalent to $G(a+h)$?


\begin{multipleChoice}
\choice {$8t^2 - 5 + a + h$}
\choice {$8 a + h^2 - 5$}
\choice [correct]{$8(a+h)^2 - 5$}
\choice {$8t^2 (a+h) - 5$}
\end{multipleChoice}

\end{question}










\begin{question}
Let $h(w) = \frac{1}{3w-4}$ \\

Let $a$ and $r$ be real numbers. \\

Which expression is equivalent to $h(a+r)$?


\begin{multipleChoice}
\choice {$\frac{1}{3 a + r - 4}$}
\choice {$\frac{1}{3w - 4 +a + r}$}
\choice {$\frac{a+r}{3w-4}$}
\choice [correct]{$\frac{1}{3(a+r)-4}$}
\end{multipleChoice}

\end{question}
