\begin{question}
Let $T(k) = -k^2 + 6k + 3$ \\

Let $F$ and $x$ be real numbers. \\

Which expression is equivalent to $T(F+x)$?


\begin{multipleChoice}
\choice {$-F+x^2 + 6F+x + 3$}
\choice [correct]{$-(F+x)^2 + 6(F+x) + 3$}
\choice {$-F^2 + x + 6F + x + 3$}
\choice {$(-k^2 + 6k + 3)(F+x)$}
\end{multipleChoice}

\end{question}
