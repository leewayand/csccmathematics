\begin{exercise}





Let $X(y) = 4 y^2 + 5$ \\

Let $h$ and $k$ be real numbers. \\





\begin{question}


Evaluate  

\[
X(y+k) = \answer{4 (y+k)^2 + 5}
\]


\end{question}








\begin{question}


Evaluate  

\[
X(y)+k = \answer{4 y^2 + 5 + k}
\]


\end{question}









\begin{question}


Evaluate  

\[
X(k \, y + h) = \answer{4 (k y+h)^2 + 5}
\]


\end{question}













\begin{question}


Evaluate  

\[
k \, X(y + h) = \answer{4 k (y+h)^2 + 5}
\]


\end{question}













\end{exercise}
