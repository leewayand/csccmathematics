\begin{exercise}





Let $M(n) = \frac{n}{n+5} + \frac{2n}{n-3}$ \\

Let $a$ and $b$ be real numbers. \\





\begin{question}


Evaluate  

\[
M(a+b) = \answer{\frac{a+b}{a+b+5} + \frac{2(a+b)}{a+b-3}}
\]


\end{question}








\begin{question}


Evaluate  

\[
M(a)+b = \answer{\frac{a}{a+5} + \frac{2a}{a-3} +b}
\]


\end{question}









\begin{question}


Evaluate  

\[
a \, M(n)+b = \answer{a(\frac{n}{n+5} + \frac{2n}{n-3})+b}
\]


\end{question}




















\end{exercise}
