\begin{exercise}





Let $M(n) = \frac{n-1}{2n+5}$ \\

Let $a$ and $b$ be real numbers. \\





\begin{question}


Evaluate  

\[
M(a+b) = \answer{\frac{a+b-1}{2(a+b)+5}}
\]


\end{question}








\begin{question}


Evaluate  

\[
M(a)+b = \answer{\frac{a-1}{2a+5}+b}
\]


\end{question}









\begin{question}


Evaluate  

\[
a \, M(n)+b = \answer{a \frac{n-1}{2n+5} + b}
\]


\end{question}




















\end{exercise}
