\begin{question}


Solve $\log_3(t+1) - \log_3(t) = 2$.








\begin{align*}
\log_3(t+1) - \log_3(t) = 2   \\
\log_3 \left( \frac{t+1}{t} \right)  & = 2  \\
\frac{t+1}{t} & = \answer{9}  \\
\answer{t+1} & = 9t  \\
1 & = 8t  \\
\frac{1}{8} & = t
\end{align*}





\end{question}
