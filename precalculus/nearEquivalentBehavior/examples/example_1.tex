\begin{question}


What is the formula for $V'(h)$ on the interval $(-6, -2)$?

\begin{multipleChoice}
\choice {$-2h-3$}
\choice {$-2h$}
\choice [correct]{$-2$}
\choice {DNE}
\end{multipleChoice}



\end{question}









\begin{question}


What is the formula for $V'(h)$ on the interval $(-2, 4)$?

\begin{multipleChoice}
\choice {$-h^2 + 9$}
\choice [correct]{$-2h$}
\choice {$2$}
\choice {$2h$}
\end{multipleChoice}



\end{question}
