The domain of $f$ is $[-9, 2) \cup (2, 8]$.\\






\begin{question} 


Classify $-9$. \\


\begin{multipleChoice}
\choice {Discontinuity}
\choice {Singularity}
\choice [correct]{Neither}
\end{multipleChoice}







Classify $-5$. \\


\begin{multipleChoice}
\choice [correct]{Discontinuity}
\choice {Singularity}
\choice {Neither}
\end{multipleChoice}






Classify $2$. \\


\begin{multipleChoice}
\choice {Discontinuity}
\choice [correct]{Singularity}
\choice {Neither}
\end{multipleChoice}




Classify $8$. \\


\begin{multipleChoice}
\choice {Discontinuity}
\choice {Singularity}
\choice [correct]{Neither}
\end{multipleChoice}




\end{question}






\begin{question}

\begin{question}

\choice [correct]{$[-2, 2)$}
\choice {$[0, 5]$}

\end{selectAll}


\end{question}
