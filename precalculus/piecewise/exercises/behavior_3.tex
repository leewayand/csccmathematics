\[
\text{Let } \, F(x) = 
\begin{cases}
  3|x+6| - 5 & [-9, -5] \\
  -2    &  (-5, -1] \\
  -(x+2)(x-4)    &  [0, 5) \\
  4    &  x = 5 \\
  2|x-7| + 2 &  (5, 9)
\end{cases}
\]

          \addplot [draw=penColor,very thick,smooth,domain=(-9:-5)] {3*abs(x+6)-5};
          \addplot [draw=penColor,very thick,smooth,domain=(-5:-1)] {-2};
          \addplot [draw=penColor,very thick,smooth,domain=(0:5)] {-(x+2)*(x-4)};

  \end{axis}
\end{tikzpicture}
\end{image}
