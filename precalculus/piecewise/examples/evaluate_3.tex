\begin{exercise}




Let $Q(t)$ be defined as follows.

\[
Q(t) = 
\begin{cases}
  \sin(t) & \text{ if }  x < 0 \\
  \cos(t) & \text{ if } 0 \leq x 
\end{cases}
\]



$Q(0) = \answer{1}$ \\


$Q\left( -\frac{\pi}{2} \right) = \answer{-1}$ \\


$Q\left( \frac{\pi}{2} \right) = \answer{0}$ \\


$Q(-\pi) = \answer{0}$ \\


$Q(\pi) = \answer{-1}$ \\




\end{exercise}
