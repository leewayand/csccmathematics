\begin{exercise}




Let $M(x)$ be defined as follows.

\[
M(x) = 
\begin{cases}
  1 - x & \text{ if } -10 < x \leq 0 \\
  x^2 & \text{ if } 0 < x \leq 7 \\
  \sqrt{x} & \text{ if } 7 < x 
\end{cases}
\]


$M(-2) = \answer{3}$ \\


$M\left( -\frac{1}{2} \right) = \answer{\frac{3}{2}}$ \\


$M(0) = \answer{1}$ \\


$M\left( \frac{1}{2} \right) = \answer{\frac{1}{4}}$ \\


$M(7) = \answer{49}$ \\


$M(8) = \answer{\sqrt{8}}$ \\


$M(100) = \answer{10}$ \\


\end{exercise}
