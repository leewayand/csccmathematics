\begin{exercise}




Let $R(k)$ be defined as follows.

\[
R(k) = 
\begin{cases}
  R(-k) & \text{ if }  k < 0 \\
  (k-1)(k-5) & \text{ if } 0 \leq k 
\end{cases}
\]

$R(k)$ is an \textbf{even} function, which means $R(a) = R(-a)$, for all real numbers, $a$. \\


\begin{question}




$R(-5) = \answer{0}$ \\


$R(-3) = \answer{-4}$ \\


$R(-1) = \answer{0}$ \\


$R(1) = \answer{0}$ \\


$R(3) = \answer{-4}$ \\


$R(5) = \answer{0}$ \\




\end{question}


\end{exercise}
