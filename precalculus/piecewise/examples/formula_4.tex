\begin{exercise}




Let $R(k)$ be defined as follows.

\[
R(k) = 
\begin{cases}
  R(-k) & \text{ if }  k < 0 \\
  (k-1)(k-5) & \text{ if } 0 \leq k 
\end{cases}
\]

$R(k)$ is an \textbf{even} function, which means $R(a) = R(-a)$, for all real numbers, $a$. \\


\begin{question}



Which of the following are zeros of $R(k)$?


\begin{selectAll}
\choice [correct]{$-5$}
\choice [correct]{$-1$}
\choice [correct]{$1$}
\choice [correct]{$5$}
\end{selectAll}

\end{question}


\end{exercise}
