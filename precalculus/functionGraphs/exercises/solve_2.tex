\begin{question} True or False

\item \wordChoice{\choice{True} \choice[correct]{False}}  There is a solution to $K(x) = 6$ \\
\item \wordChoice{\choice[correct]{True} \choice{False}}  There is a solution to $P(w) = -4$ \\

\end{question}





\begin{question} Solutions

$P(w) = 4$ has $\answer{2}$ solutions.

\end{question}




\begin{question} Solutions

\end{question}

\item \wordChoice{\choice[correct]{True} \choice{False}}  There is a solution to $P(w) < 0$ \\
