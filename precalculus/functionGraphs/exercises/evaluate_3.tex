\begin{question}
Evaluate the following:

\begin{itemize}
\item $P(-6) + K(6) = \answer[tolerance=0.1]{2}$ \\
\item $K(2) - K(-2) = \answer[tolerance=0.1]{4}$ \\
\item $P(0) - K(5) = \answer[tolerance=0.1]{-7}$ \\
\end{itemize}
\end{question}






\begin{question}

\item $P(-6+9) + K(-3+5) = \answer[tolerance=0.1]{-2}$ \\
\item $2 K(-1) - 2 P(1)  = \answer[tolerance=0.1]{18}$ \\
\item $P(3-8) K(-2-2) = \answer[tolerance=0.1]{0}$ \\
\end{itemize}
\end{question}
