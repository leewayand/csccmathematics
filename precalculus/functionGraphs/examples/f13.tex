\begin{question} 


What is the domain of $F$?\\


\begin{multipleChoice}
\choice {$(-\infty, \infty)$}
\choice {$[-9.9]$}
\choice {$(-9,9)$}
\choice [correct]{$(-\infty, 3) \cup (3, \infty)$}
\end{multipleChoice}

\end{question}






\begin{question} 


What is the range of $F$?\\


\begin{multipleChoice}
\choice {$(-\infty, \infty)$}
\choice {$[-10.10]$}
\choice {$(-10,10)$}
\choice [correct]{$(-\infty, 0) \cup (0, \infty)$}
\end{multipleChoice}


\end{question}
