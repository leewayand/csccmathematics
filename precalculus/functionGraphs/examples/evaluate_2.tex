\begin{question}
Evaluate the following:

\begin{itemize}
\item $P(-6) = \answer[tolerance=0.1]{2}$ \\
\item $P(1) = \answer[tolerance=0.1]{-6}$ \\
\item $P(-7) = \answer[tolerance=0.1]{5}$ \\
\end{itemize}
\end{question}






\begin{question}
Evaluate the following:

\begin{itemize}
\item $P(-6) + P(4) = \answer[tolerance=0.1]{2}$ \\
\item $P(-6+4) = \answer[tolerance=0.1]{-1}$ \\
\item $P(1) + P(2)  = \answer[tolerance=0.1]{-10}$ \\
\item $P(1+2) = \answer[tolerance=0.1]{-2}$ \\
\end{itemize}
\end{question}
