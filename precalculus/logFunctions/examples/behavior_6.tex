\begin{exercise}






Let $L(z) = -\log_3(5-z) + 1$ with its natural domain. \\





\begin{question}


The graph of $y = L(z)$ is to the \wordChoice{\choice[correct]{left} \choice{right}} of the vertical asymptote $z = 5$.


\end{question}





\begin{question}


The domain of $y = L(z)$ is 

\[
\left( \answer{-\infty}, \answer{5} \right)
\]


\end{question}






\begin{question}


$L(z)$ is an \wordChoice{\choice[correct]{increasing} \choice{decreasing}} function.


\end{question}











\begin{question}


\[
\lim\limits_{z \to 5^-} L(z) = \answer{\infty}
\]


\end{question}








\begin{question}


\[
\lim\limits_{z \to -\infty} L(z) = \answer{-\infty}
\]


\end{question}










\end{exercise}
