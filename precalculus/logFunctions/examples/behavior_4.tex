\begin{exercise}






Let $K(t) = \log_8(4-t) + 3$ with its natural domain. \\





\begin{question}


The graph of $z = K(t)$ is to the \wordChoice{\choice[correct]{left} \choice{right}} of the vertical asymptote $t = 4$.


\end{question}





\begin{question}


The domain of $z = K(t)$ is 

\[
\left( \answer{-\infty}, \answer{4} \right)
\]


\end{question}






\begin{question}


$K(t)$ is an \wordChoice{\choice{increasing} \choice[correct]{decreasing}} function.


\end{question}











\begin{question}


\[
\lim\limits_{t \to 4^-} K(t) = \answer{-\infty}
\]


\end{question}








\begin{question}


\[
\lim\limits_{t \to -\infty} K(t) = \answer{\infty}
\]


\end{question}










\end{exercise}
