\begin{exercise}






Let $g(x) = -2 \log_3(x-5)$ with its natural domain. \\





\begin{question}


The graph of $y = g(x)$ is to the \wordChoice{\choice{left} \choice[correct]{right}} of the vertical asymptote $x = 5$.


\end{question}





\begin{question}


The domain of $y = g(x)$ is 

\[
\left( \answer{5}, \answer{\infty} \right)
\]


\end{question}






\begin{question}


$g(x)$ is an \wordChoice{\choice{increasing} \choice[correct]{decreasing}} function.


\end{question}











\begin{question}


\[
\lim\limits_{x \to 5^+} g(x) = \answer{\infty}
\]


\end{question}








\begin{question}


\[
\lim\limits_{x \to \infty} g(x) = \answer{-\infty}
\]


\end{question}










\end{exercise}
