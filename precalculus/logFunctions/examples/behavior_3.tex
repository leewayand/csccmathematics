\begin{exercise}






Let $h(x) = -2 \log_3(5-x)$ with its natural domain. \\





\begin{question}


The graph of $y = h(x)$ is to the \wordChoice{\choice[correct]{left} \choice{right}} of the vertical asymptote $x = 5$.


\end{question}





\begin{question}


The domain of $y = h(x)$ is 

\[
\left( \answer{-\infty}, \answer{5} \right)
\]


\end{question}






\begin{question}


$h(x)$ is an \wordChoice{\choice[correct]{increasing} \choice{decreasing}} function.


\end{question}











\begin{question}


\[
\lim\limits_{x \to 5^-} h(x) = \answer{\infty}
\]


\end{question}








\begin{question}


\[
\lim\limits_{x \to -\infty} h(x) = \answer{-\infty}
\]


\end{question}










\end{exercise}
