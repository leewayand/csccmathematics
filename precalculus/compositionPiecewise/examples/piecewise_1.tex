\begin{exercise}





\[
T(k) = 
\begin{cases}
  2k + 4 &  (-\infty, -4]   \\
  -k - 5 &  (-4, 7]  \\
  k - 6 & (7, \infty)
\end{cases}
\]












\[
B(w) = 
\begin{cases}
  w - 8 &   -\infty < w < 1 \\
  -2w + 5 &  1 \leq w < \infty
\end{cases}
\]










\begin{question}



Use the formulas above to evaluate the following expressions.



\[
\begin{array}{l|l}
T(5) = \answer{-10}  & B(T(5)) = \answer{-18}   \\
B(1) = \answer{3}  & B(B(1)) = \answer{-1}   \\
B(4) = \answer{-3}  & T(B(4)) = \answer{-2}   \\
T(-3) = \answer{-2}  & T(T(-3)) = \answer{-3}   
\end{array}
\]




\end{question}














\end{exercise}
