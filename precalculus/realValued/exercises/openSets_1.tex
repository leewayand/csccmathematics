\begin{question}
$(-5, 2)$ is an open interval and $-1 \in (-5, 2)$.  Select all of the open intervals inside $(-5, 2)$ that contain $-1$.
  \begin{selectAll}
  \choice [correct]{$(-2, 0)$}
  \choice [correct]{$(-3, 0)$}
  \choice [correct]{$(-1.5, 1)$}
  \choice {$(0, 1.7)$}
  \end{selectAll}
\end{question}
















\begin{question}
$(-3, 0) \cup (0, 3) \cup (6, 7)$ is an open set and $-1 \in (-3, 0) \cup (0, 3) \cup (6, 7)$.  Select all of the open intervals inside $(-3, 0) \cup (0, 3) \cup (6, 7)$ that contain $-1$.
  \begin{selectAll}
  \choice [correct]{$(-2, 0)$}
  \choice {$(-2, 1)$}
  \choice {$(-1, 0)$}
  \choice [correct]{$(-1.5, -0.5)$}
  \end{selectAll}
\end{question}









\begin{question}

  \begin{selectAll}

  \choice {Closed}
  \choice [correct]{Neither}
  \end{selectAll}
\end{question}
