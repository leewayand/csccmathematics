\begin{exercise}




Let $G(w) = 5 \cdot 2^{3-w} + 4$ with its natural domain.


\begin{question}


$G$ is \wordChoice{\choice{increasing} \choice[correct]{decreasing}} on its domain. \\


\end{question}







\begin{question}


$G$ becomes unbounded as $w$ approaches $\answer{-\infty}$.


\end{question}





\begin{question}


$\left( 3, \answer{9} \right) $is an ``anchor point" of the graph of $G$. \\


\end{question}








\begin{question}


The graph of $y=G(w)$ has a horizontal asymptote of

\begin{multipleChoice}
\choice{$y=0$}
\choice{$x=0$}
\choice[correct]{$y=4$}
\choice{$x=4$}
\end{multipleChoice}


\end{question}








\end{exercise}
