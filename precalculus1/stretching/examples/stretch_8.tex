\end{cases}
\]

$V$ has many features and characteristics. \\






\begin{question}


The domain of $V$ is 

\[
\left[ \answer{-6}, \answer{6} \right)
\]


\end{question}








\begin{question}



Let $g(t) = 4 V(2t - 5) - 1$.



The domain of $g$ is 

\[
\left[ \answer{\frac{-1}{2}}, \answer{\frac{11}{2}} \right)
\]




\end{question}












\begin{question}



Let $H(b) = -5 V(7 - 3t) + 2$.



The domain of $H$ is 

\[
\left( \answer{\frac{1}{3}}, \answer{\frac{13}{2}} \right]
\]




\end{question}
