\begin{question}

Which of these formulas define an increasing function on its natural domain? \\



\begin{selectAll}
\choice [correct]{$f(A) = 3 \, x + 4$}
\choice {$g(n) = -3 \, n + 4$}
\choice [correct]{$h(k) = -3 \, (4 - k)$}
\choice [correct]{$W(y) = \frac{1}{3} \, y + 4$}
\choice {$T(d) = \frac{-2}{3} \, d + 4$}
\choice [correct]{$c(v) = \frac{-2}{3} \, (-v) + 4$}
\choice {$p(B) = 3 \, (4 - B))$}
\choice [correct]{$Q(N) = -3 \, (x + 4) + 5(N + 4)$}
\choice {$M(T) = 3 \, (x + 4) - 5(N + 4)$}
\end{selectAll}




\end{question}
