\begin{exercise}





Let 


\[
K(h) = -4 \cdot e^{2h + 1} +7
\]





\begin{question}



The zero of $K$ is


\[
\answer{\frac{1}{2} (ln(\frac{7}{4}) - 1)}
\]


\end{question}









\begin{question}



$V$ is positive on 


\[
\left( \answer{-\infty}, \answer{\frac{1}{2} (ln(\frac{7}{4}) - 1)}  \right)
\]


\end{question}








\begin{question}



$V$ is negative on 


\[
\left( \answer{\frac{1}{2} (ln(\frac{7}{4}) - 1)}, \answer{\infty}  \right)
\]


\end{question}
