\begin{exercise}




Let $T(y) = -3 \cdot 5^{y+1} - 4$ with its natural domain.


\begin{question}


$T$ is \wordChoice{\choice{increasing} \choice[correct]{decreasing}} on its domain. \\


\end{question}







\begin{question}


$T$ becomes unbounded as $y$ approaches $\answer{\infty}$.


\end{question}





\begin{question}


$\left( -1, \answer{7} \right) $is an ``anchor point" of the graph of $T$. \\


\end{question}








\begin{question}


The graph of $z=T(y)$ has a horizontal asymptote of

\begin{multipleChoice}
\choice{$y=0$}
\choice{$x=0$}
\choice{$y=4$}
\choice[correct]{$z=4$}
\choice{$x=4$}
\end{multipleChoice}


\end{question}








\end{exercise}
