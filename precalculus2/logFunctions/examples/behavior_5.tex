\begin{exercise}






Let $X(t) = log_5(t+3) - 4$ with its natural domain. \\





\begin{question}


The graph of $z = X(t)$ is to the \wordChoice{\choice{left} \choice[correct]{right}} of the vertical asymptote $t = -3$.


\end{question}





\begin{question}


The domain of $z = X(t)$ is 

\[
\left( \answer{-3}, \answer{\infty} \right)
\]


\end{question}






\begin{question}


$X(t)$ is an \wordChoice{\choice[correct]{increasing} \choice{decreasing}} function.


\end{question}











\begin{question}


\[
\lim\limits_{t \to -3^+} X(t) = \answer{-\infty}
\]


\end{question}








\begin{question}


\[
\lim\limits_{t \to \infty} X(t) = \answer{\infty}
\]


\end{question}










\end{exercise}
