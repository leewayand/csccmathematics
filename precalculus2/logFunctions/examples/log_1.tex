\begin{question}

What is $log_5(25)$ ?

\begin{multipleChoice}
\choice{$log_5(25)$ is the thing you raise $25$ to, to get $5$}
\choice{$log_5(25)$ is the thing you raise $10$ to, to get $35$}
\choice[correct]{$log_5(25)$ is the thing you raise $5$ to, to get $25$}
\choice{$log_5(25)$ is the thing you raise $e$ to, to get $25$}
\end{multipleChoice}


\end{question}






\begin{question}


$log_5(25) = \answer{2}$.


\end{question}
