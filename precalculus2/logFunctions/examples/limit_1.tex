Let $Z(t) = -log_7(t+4) - 3$ with its natural domain. \\







\begin{question}


\[
\lim\limits_{t \to -4^+} Z(t) = \answer{\infty}
\]


\end{question}








\begin{question}


\[
\lim\limits_{t \to \infty} Z(t) = \answer{-\infty}
\]


\end{question}
