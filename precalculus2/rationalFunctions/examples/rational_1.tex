\begin{exercise}




Let $g(x)$ be a rational function defined by the formula

\[
g(x) = \frac{7(x-5)(x+1)^2}{(x-3)(x+4)}
\]


\begin{question}


Select all of the roots of $g(x)$.

\begin{selectAll}
\choice[correct]{$5$}
\choice[correct]{$-1$}
\choice{$3$}
\choice{$-4$}
\end{selectAll}

\end{question}






\begin{question}


Select all of the singularities of $g(x)$.

\begin{selectAll}
\choice{$5$}
\choice{$-1$}
\choice[correct]{$3$}
\choice[correct]{$-4$}
\end{selectAll}

\end{question}










\begin{question}


Select all of the vertical asymptotes on the graph of $y = g(x)$.

\begin{selectAll}
\choice[correct]{$x = 3$}
\choice{$y = 3$}
\choice{$y = 7$}
\choice[correct]{$x = -4$}
\choice{$y = -4$}
\choice{$x = 7$}
\end{selectAll}

\end{question}












\begin{question}


Select all of the horizontal asymptotes on the graph of $y = g(x)$.

\begin{selectAll}
\choice{$x = 3$}
\choice{$y = 3$}
\choice[correct]{$y = 7$}
\choice{$x = -4$}
\choice{$y = -4$}
\choice{$x = 7$}
\end{selectAll}

\end{question}












\begin{question}


$-4$ is a singularity for $g$.

\begin{multipleChoice}
\choice[correct]{True}
\choice{False}
\end{multipleChoice}




$-4$ is a removeable singularity for $g$.

\begin{multipleChoice}
\choice{True}
\choice[correct]{False}
\end{multipleChoice}



\end{question}















\end{exercise}
